Try and catch --> main target //c++ for Error Handling 1.Compile time error 2.Runtime time error Division by Zero Handling javascript Copy Edit function divideNumbers(a, b) { try { if (b === 0) { throw new Error("Cannot divide by zero!"); } console.log("Result:", a / b); } catch (error) { console.log("Error:", error.message); } } // Test cases divideNumbers(10, 2); // Outputs: Result: 5 divideNumbers(10, 0); // Outputs: Error: Cannot divide by zero! Handling Invalid Number Conversion javascript Copy Edit function convertToNumber(input) { try { let number = Number(input); if (isNaN(number)) { throw new Error("Invalid number input!"); } console.log("Converted number:", number); } catch (error) { console.log("Error:", error.message); } } // Test cases convertToNumber("42"); // Outputs: Converted number: 42 convertToNumber("hello"); // Outputs: Error: Invalid number input! Accessing Array Elements Safely javascript Copy Edit function getArrayElement(arr, index) { try { if (!Array.isArray(arr) || arr.length === 0 || index < 0 || index >= arr.length) { throw new Error("Error: Invalid array or index out of range!"); } console.log("Element at index", index, ":", arr[index]); } catch (error) { console.log(error.message); } } // Test cases getArrayElement([10, 20, 30], 1); // Outputs: Element at index 1: 20 getArrayElement([10, 20, 30], 5); // Outputs: Error: Invalid array or index out of range! getArrayElement(null, 1); // Outputs: Error: Invalid array or index out of range! getArrayElement([], 0); // Outputs: Error: Invalid array or index out of range!
